// roadmap.go — 技术路线图框架
package leadership

import (
	"time"
)

// TechRoadmap 技术路线图
type TechRoadmap struct {
	Name        string            // 路线图名称
	Vision      string            // 技术愿景 (Why)
	Objectives  []TechObjective   // 技术目标 (What)
	Track       []TechTrack       // 技术轨道 (How)
	Timeline    Timeline          // 时间线
	Stakeholders map[string]string // 利益相关者
	SuccessMetrics map[string]int   // 成功指标
}

// TechObjective 技术目标 (OKR 格式)
type TechObjective struct {
	Objective string   // 目标描述
	KeyResults []KeyResult // 关键结果
	Quarter     string    // 季度 (Q3 2025)
	Priority    Priority  // 优先级
}

// KeyResult 关键结果
type KeyResult struct {
	Description string    // 描述
	Metric      string    // 衡量指标
	TargetValue string    // 目标值
	Deadline    time.Time // 截止日期
	Status      string    // 进行中/已完成/延期
}

// TechTrack 技术轨道
type TechTrack struct {
	Name        string    // 轨道名称
	Description string    // 描述
	Projects    []Project // 项目列表
	Owner       string    // 负责人
	Dependencies []string  // 依赖项
	Milestones   []Milestone // 里程碑
}

// Project 项目
type Project struct {
	ID         string
	Name       string
	Description string
	StartDate  time.Time
	EndDate    time.Time
	Status     string // Planning / InProgress / Completed
	Priority   string // P0 / P1 / P2
}

// Milestone 里程碑
type Milestone struct {
	Name       string
	TargetDate time.Time
	Description string
	Completed  bool
}

// 制定技术路线图五步法:
//
// Step 1: 理解业务战略
// ├── 与公司/业务战略对齐

// ├── 识别业务痛点
// └── 确定技术如何驱动业务价值
//
// Step 2: 评估当前状态
// ├── 技术现状评估 (架构/性能/质量/安全)
// ├── 团队能力评估
// ├── 技术债务审计
// └── 行业对标分析
//
// Step 3: 确定技术方向
// ├── 技术趋势分析 (AI/Cloud/Microservices)
// ├── 技术选型决策 (新框架/新语言/新平台)
// ├── 架构演进方向 (单体→微服务/混合云/Serverless)
// └── 创新机会识别
//
// Step 4: 制定路线图
// ├── 短期 (1-3月): 紧急/高价值
// ├── 中期 (3-6月): 中期优化
// ├── 长期 (6-12月): 战略性投资

### 3.1 架构决策记录 (ADR) — 源码级

```go
// adr.go — 架构决策记录
package leadership

import (
	"time"
)

// ADRStatus 决策状态
type ADRStatus string

const (
	ADRStatusProposed    ADRStatus = "PROPOSED"   // 提议中
	ADRStatusAccepted    ADRStatus = "ACCEPTED"   // 已接受
	ADRStatusSuperseded  ADRStatus = "SUPERSEDED" // 被取代
	ADRStatusDeprecated  ADRStatus = "DEPRECATED" // 已废弃
)

// ADR 架构决策记录 (Architecture Decision Record)
type ADR struct {
	ID            string        `json:"id"`             // ADR-001
	Title         string        `json:"title"`          // 决策标题
	Status        ADRStatus     `json:"status"`         // 状态
	DecisionDate  time.Time     `json:"decision_date"`  // 决策日期
	Decider       string        `json:"decider"`        // 决策者
	Context       string        `json:"context"`        // 背景 (为什么需要决策)
	Options       []ADROption   `json:"options"`        // 备选方案
	Decision      string        `json:"decision"`       // 最终决策
	Rationale     string        `json:"rationale"`      // 决策理由
	Consequences  ADREffect     `json:"consequences"`   // 后果 (Pros/Cons)
	Dependencies  []string      `json:"dependencies"`   // 依赖的 ADR
	Related       []string      `json:"related"`        // 相关的 ADR
	CreationDate  time.Time     `json:"creation_date"`  // 创建日期
	LastUpdated   time.Time     `json:"last_updated"`   // 更新日期
	Reviewers     []string      `json:"reviewers"`      // 评审人员
}

// ADROption 备选方案
type ADROption struct {
	Name     string `json:"name"`       // 方案名称
	Pros     []string `json:"pros"`     // 优点
	Cons     []string `json:"cons"`     // 缺点
	Cost     string `json:"cost"`     // 成本估算
	Risk     string `json:"risk"`     // 风险等级
	Complexity string `json:"complexity"` // 实现复杂度
}

// ADREffect 决策后果
type ADREffect struct {
	Positive []string `json:"positive"`  // 积极影响
	Negative []string `json:"negative"`  // 消极影响
	Mitigation []string `json:"mitigation"` // 缓解措施
}

// ADR 模板 (Markdown):
// # ADR-001: [标题]
//
// ## 状态: PROPOSED | ACCEPTED | SUPERSEDED | DEPRECATED
// ## 日期: 2025-06-10
// ## 决策者: Ryan
//
// ## 上下文
//
// [描述当前的问题/背景]
//
// ## 备选方案
//
// ### 方案 1: [名称]
// - 优点:
//   - [优点]
// - 缺点:
//   - [缺点]
//
// ### 方案 2: [名称]
// - 优点:
//   - [优点]
// - 缺点:
//   - [缺点]
//
// ## 决策
//
// 我们选择: [方案 X]
//
// ## 理由
//
// [解释为什么选择这个方案]
//
// ## 后果
//
// 积极:
// - [积极影响]
//
// 消极:
// - [消极影响]
//
// 缓解:
// - [缓解措施]
//
// ## 相关
//
// - [相关 ADR 链接]

// 常用 ADR 模板库:
// ADR-001: 技术选型 (微服务 vs 单体)

	DebtTypeDocumentation TechDebtType = "DOC"          // 文档债务
	DebtTypeInfra         TechDebtType = "INFRA"        // 基础设施债务
	DebtTypeKnowledge     TechDebtType = "KNOWLEDGE"    // 知识债务 (bus factor)
)

// TechDebt 技术债项
type TechDebt struct {
	ID           string        `json:"id"`            // TD-001
	Title        string        `json:"title"`         // 标题
	Description  string        `json:"description"`   // 描述
	Type         TechDebtType  `json:"type"`          // 类型
	Impact       DebtImpact    `json:"impact"`        // 影响
	RiskLevel    string        `json:"risk_level"`    // P0/P1/P2/P3
	EstimatedEffort string      `json:"estimated_effort"` // 修复工时
	CreatedDate  time.Time     `json:"created_date"`  // 发现日期
	AssignedTo   string        `json:"assigned_to"`   // 负责人
	Status       string        `json:"status"`        // Open/In Progress/Resolved/Won't Fix
	CreatedBy    string        `json:"created_by"`    // 发现者
}

// DebtImpact 技术债影响
type DebtImpact struct {
	Performance    float64 `json:"performance"`  // 性能影响 (0-10)
	Maintainability float64 `json:"maintainability"` // 可维护性 (0-10)
	Security     float64 `json:"security"`    // 安全风险 (0-10)
	DevelopmentSpeed float64 `json:"dev_speed"` // 开发速度影响 (0-10)
}

// TechDebtScore 技术债评分
func (d *TechDebt) Score() float64 {
	// 技术债评分 = 影响 × 风险 × 成本
	riskMap := map[string]float64{"P0": 4, "P1": 3, "P2": 2, "P3": 1}
	riskFactor := riskMap[d.RiskLevel]
	
	avgImpact := (d.Impact.Performance + d.Impact.Maintainability + 
		d.Impact.Security + d.Impact.DevelopmentSpeed) / 4
		
	// 成本系数 (根据修复工时)
	costFactor := 1.0
	switch d.EstimatedEffort {
	case "< 1天":
		costFactor = 0.5
	case "1-3天":
		costFactor = 1.0
	case "3-7天":
		costFactor = 2.0
	case "1-2周":
		costFactor = 3.0
	case "> 2周":
		costFactor = 4.0
	}
	
	return avgImpact * riskFactor * costFactor
}

// 技术债管理四步法:
//
// Step 1: 识别 (Identify)
// ├── 代码扫描 (SonarQube / Go static analysis)
// ├── 性能分析 (pprof / benchmark)
// ├── 安全扫描 (CVE / dependency audit)
// ├── 测试覆盖率分析
// └── 架构评审
//
// Step 2: 评估 (Evaluate)
// ├── 技术债评分 (Impact × Risk × Effort)
// ├── 优先级排序 (P0/P1/P2/P3)
// ├── 成本估算 (修复工时)
// └── 风险评估 (不修复的后果)
//
// Step 3: 计划 (Plan)
// ├── P0: 立即修复 (嵌入 Sprint)
// ├── P1: 短期修复 (1-2 Sprint)
// ├── P2: 中期修复 (1 个 Quarter)
// └── P3: 长期修复 (技术预研)
//
// Step 4: 追踪 (Track)
// ├── 定期回顾 (每周 Sprint Review)
// ├── 持续监控 (自动化扫描)
// ├── 指标追踪 (技术债趋势)
// └── 定期报告 (向管理层汇报)

// 技术债仪表盘指标:
// 技术债总数: 当前待修复的技术债项数
// 技术债总额: 所有技术债项的评分总和
// 技术债趋势: 每月新增 vs 修复的技术债
// P0 占比: P0 级别技术债占总数的百分比
// 平均修复时间: 从发现到修复的平均时间
// 技术债密度: 每 1000 行代码的技术债数
```

### Q2：如何管理和量化技术债？

<details>
<summary>点击查看参考答案</summary>

**四步法**:

1. **识别**: 静态分析 + 性能分析 + 安全扫描 + 代码审查
2. **评估**: Impact × Risk × Effort → 评分排序
3. **计划**: P0(立即) P1(1-2 Sprint) P2(1 Quarter) P3(长期)
4. **追踪**: 每周回顾 + 自动化监控 + 指标报告

**评分模型**:
```
评分 = 平均影响(0-10) × 风险等级(P0=4,P1=3,P2=2,P3=1) × 成本系数
```

**仪表盘指标**: 总数/总额/趋势/P0占比/平均修复时间/技术债密度
</details>

### Q3：技术债和特性开发如何平衡？

<details>
<summary>点击查看参考答案</summary>

**原则**: "技术债不是敌人, 而是杠杆。关键在于知道欠多少, 怎么还。"

**策略**:
1. **Sprint 预留**: 每个 Sprint 预留 10-20% 时间处理技术债
2. **技术债 Sprint**: 每季度专门 1 个 Sprint 治理技术债
3. **嵌入式**: P0 技术债直接嵌入特性开发
4. **量化驱动**: 用数据证明技术债对业务的影响 (开发速度下降/bug 增多)
5. **技术债看板**: 透明化管理, 团队可见

**平衡公式**:
- 纯特性: 80-90% + 技术债 10-20% (常规)
- 重构期: 特性 50% + 技术债 50% (特殊 Sprint)
- 紧急期: 特性 100% + 技术债 0% (短期, 需记录债务)
</details>

### 3.2 技术债量化 (技术债度量模型)

```go
// tech_debt.go — 技术债量化模型 (Go 实现)
package leadership

import (
	"fmt"
	"strings"
	"time"
)

// DebtType 技术债类型
type DebtType string

const (
	DebtCodeQuality   DebtType = "code_quality"    // 代码质量
	DebtDocumentation  DebtType = "documentation"   // 文档缺失
	DebtTesting        DebtType = "testing"         // 测试不足
	DebtArchitecture   DebtType = "architecture"    // 架构问题
	DebtSecurity       DebtType = "security"        // 安全风险
	DebtPerformance    DebtType = "performance"     // 性能问题
	DebtScalability    DebtType = "scalability"     // 扩展性问题
	DebtCompliance     DebtType = "compliance"      // 合规问题
)

// TechnicalDebt 技术债项
type TechnicalDebt struct {
	ID          string     `json:"id"`
	Title       string     `json:"title"`
	Type        DebtType   `json:"type"`
	Description string     `json:"description"`
	Priority    DebtPriority `json:"priority"` // critical/high/medium/low
	RiskScore   float64    `json:"risk_score"` // 0-10
	ImpactScore float64    `json:"impact_score"` // 0-10
	CostScore   float64    `json:"cost_score"`   // 0-10: 修复成本 (越大越贵)
	AgeMonths   int        `json:"age_months"`   // 存在时间 (月)
	Owner       string     `json:"owner"`
	Status      string     `json:"status"` // open/in_progress/resolved
	CreatedAt   time.Time  `json:"created_at"`
	UpdatedAt   time.Time  `json:"updated_at"`
}

// DebtPriority 技术债优先级
type DebtPriority string

const (
	PriorityCritical DebtPriority = "critical" // 立即修复
	PriorityHigh     DebtPriority = "high"     // 下一个迭代
	PriorityMedium   DebtPriority = "medium"   // 本季度
	PriorityLow      DebtPriority = "low"      // 有空再做
)

// DebtRiskScore 技术债风险评分 (0-100)
// 公式: Risk = RiskScore × (1 + AgeMultiplier) × ImpactMultiplier
// RiskScore = 0.3 × Risk + 0.4 × Impact + 0.3 × Cost
func (d *TechnicalDebt) CalculateRiskScore() float64 {
	baseRisk := 0.3*d.RiskScore + 0.4*d.ImpactScore + 0.3*d.CostScore
	ageMultiplier := 1.0 + float64(d.AgeMonths)/12.0 // 每过一年增加 100%
	return baseRisk * ageMultiplier
}

// DebtManager 技术债管理器
type DebtManager struct {
	debtList []*TechnicalDebt
}

// AddDebt 添加技术债项
func (m *DebtManager) AddDebt(debt *TechnicalDebt) {
	m.debtList = append(m.debtList, debt)
}

// GetTopDebt 获取风险最高的 N 个技术债
func (m *DebtManager) GetTopDebt(n int) []*TechnicalDebt {
	// 计算所有技术债的风险评分
	type debtWithRisk struct {
		debt    *TechnicalDebt
		risk    float64
	}
	
	type debtWithRisk {
		debt    *TechnicalDebt
		risk    float64
	}
	
	var scored []*debtWithRisk
	for _, debt := range m.debtList {
		risk := debt.CalculateRiskScore()
		scored = append(scored, &debtWithRisk{debt, risk})
	}
	
	// 按风险评分降序排序
	sort.Slice(scored, func(i, j int) bool {
		return scored[i].risk > scored[j].risk
	})
	
	// 返回前 N 个
	if n > len(scored) {
		n = len(scored)
	}
	
	var result []*TechnicalDebt
	for i := 0; i < n; i++ {
		result = append(result, scored[i].debt)
	}
	
	return result
}

// GetDebtByType 按类型分组统计
func (m *DebtManager) GetDebtByType() map[DebtType]int {
	counts := make(map[DebtType]int)
	for _, debt := range m.debtList {
		counts[debt.Type]++
	}
	return counts
}
```

## 第五部分：自测

### Q1：技术债的 "Reckless" 和 "Pragmatic" 有什么区别？

<details>
<summary>点击查看参考答案</summary>

**Reckless ( recklessness ) 债**:
- 明知有问题但不处理
- 没有计划, 没有文档, 没有优先级
- 团队默认 "以后再说"
- 风险: 高 (随时可能崩溃)
- 处理: 立即列入最高优先级

**Pragmatic (pragmatic) 债**:
- 权衡后的选择, 有意为之
- 有明确的偿还计划
- 用于快速获取 MVP
- 风险: 低 (明确计划偿还)
- 处理: 按计划偿还

**关键区别**: Pragmatic 是 "有计划的债", Reckless 是 "无计划的债"。
</details>

### Q2：ADR 的核心价值是什么？为什么不能只在口头或 Slack 中讨论架构决策？

<details>
<summary>点击查看参考答案</summary>

**ADR 的核心价值**:

1. **可追溯性**: 为什么做了这个决策? 谁做的? 什么时候?
2. **知识沉淀**: 新成员能快速了解历史决策
3. **责任明确**: 决策者明确, 避免事后推诿
4. **上下文完整**: 包含替代方案, 未来回顾时有完整视角
5. **决策质量**: 书面化倒逼思考更充分

**口头/Slack 的局限**:
- 信息散乱, 难以搜索
- 无结构化, 缺少上下文
- 无法追踪决策状态
- 团队成员理解不一致

**最佳实践**:
- 重要决策必须写 ADR
- ADR 存入版本控制系统 (Git)
- Review 流程中检查是否有 ADR
- 定期 Review 已接受的 ADR 是否仍然适用
</details>

### Q3：技术债管理的 "20% 法则" 是什么意思？

<details>
<summary>点击查看参考答案</summary>

**20% 法则**: 每个 Sprint 至少预留 20% 的容量用于技术债偿还。

**为什么是 20%**:
- 低于 20%: 技术债会加速积累
- 高于 20%: 可能影响业务交付
- 20% 是一个经验值, 可根据团队情况调整

**实施方法**:
1. 在每个 Sprint Planning 中预留 20% 容量
2. 技术债项按优先级排序
3. 优先处理 Critical 和 High 优先级
4. 每月 Review 技术债偿还进度
5. 调整 20% 比例 (根据业务压力和技术债积累速度)

**注意**: 20% 是底线, 如果技术债严重, 可能需要更高比例。
</details>
